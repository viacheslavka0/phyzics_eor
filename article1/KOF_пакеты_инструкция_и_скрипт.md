def _set_run_font(run, size=14, bold=False, italic=False):
    run.font.name = "Times New Roman"
    run.font.size = Pt(size)
    run.bold = bold
    run.italic = italic
    rpr = run._element.get_or_add_rPr()
    rfonts = rpr.find(qn("w:rFonts"))
    if rfonts is None:
        rfonts = OxmlElement("w:rFonts")
        rpr.append(rfonts)
    for k in (qn("w:ascii"), qn("w:hAnsi"), qn("w:cs"), qn("w:eastAsia")):
        rfonts.set(k, "Times New Roman")


def setup_kof(doc: Document) -> None:
    sec = doc.sections[0]
    sec.top_margin = Mm(20)
    sec.bottom_margin = Mm(20)
    sec.left_margin = Mm(20)
    sec.right_margin = Mm(20)
    st = doc.styles["Normal"]
    st.font.name = "Times New Roman"
    st.font.size = Pt(14)
    rpr = st.element.get_or_add_rPr()
    rf = rpr.find(qn("w:rFonts"))
    if rf is None:
        rf = OxmlElement("w:rFonts")
        rpr.append(rf)
    for k in (qn("w:ascii"), qn("w:hAnsi"), qn("w:cs"), qn("w:eastAsia")):
        rf.set(k, "Times New Roman")
    pf = st.paragraph_format
    pf.line_spacing = 1.5
    pf.space_before = Pt(0)
    pf.space_after = Pt(0)
    pf.first_line_indent = Cm(1.0)
    pf.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY


def p_blank(doc):
    doc.add_paragraph()


def p_right_bold_lines(doc, lines, size=14):
    for t in lines:
        p = doc.add_paragraph()
        p.alignment = WD_ALIGN_PARAGRAPH.RIGHT
        p.paragraph_format.first_line_indent = Cm(0)
        r = p.add_run(t)
        _set_run_font(r, size=size, bold=True, italic=False)


def p_title_caps(doc, title_ru: str):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    p.paragraph_format.first_line_indent = Cm(0)
    p.paragraph_format.space_before = Pt(0)
    p.paragraph_format.space_after = Pt(6)
    r = p.add_run(title_ru.upper())
    _set_run_font(r, size=14, bold=True)


def p_abs_kw(doc, label: str, text: str):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p.paragraph_format.first_line_indent = Cm(0)
    lab = p.add_run(label)
    _set_run_font(lab, bold=True)
    body = p.add_run(" " + text)
    _set_run_font(body, bold=False)


def p_section(doc, title: str):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.LEFT
    p.paragraph_format.first_line_indent = Cm(0)
    p.paragraph_format.space_before = Pt(10)
    p.paragraph_format.space_after = Pt(4)
    r = p.add_run(title)
    _set_run_font(r, bold=True)


def p_body(doc, text: str):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    p.paragraph_format.first_line_indent = Cm(1.0)
    r = p.add_run(text)
    _set_run_font(r)


def p_refs_heading(doc):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    p.paragraph_format.first_line_indent = Cm(0)
    p.paragraph_format.space_before = Pt(12)
    r = p.add_run("Список литературы")
    _set_run_font(r, bold=True)


def p_ref_item(doc, n: int, text: str):
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    pf = p.paragraph_format
    pf.left_indent = Cm(0.75)
    pf.first_line_indent = Cm(-0.75)
    r = p.add_run(f"{n}. {text}")
    _set_run_font(r, size=12)


def add_figure(doc, path: Path, caption: str, width_cm: float = 14.5):
    p_blank(doc)
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    p.paragraph_format.first_line_indent = Cm(0)
    run = p.add_run()
    if path.exists():
        run.add_picture(str(path), width=Cm(width_cm))
    p2 = doc.add_paragraph()
    p2.alignment = WD_ALIGN_PARAGRAPH.CENTER
    p2.paragraph_format.first_line_indent = Cm(0)
    r2 = p2.add_run(caption)
    _set_run_font(r2, bold=True, italic=False)
    p_blank(doc)


def add_table_kof(doc, num: int, title: str, headers, rows):
    p_blank(doc)
    p_num = doc.add_paragraph()
    p_num.alignment = WD_ALIGN_PARAGRAPH.RIGHT
    p_num.paragraph_format.first_line_indent = Cm(0)
    p_num.add_run(f"Таблица {num}").bold = True
    for r in p_num.runs:
        _set_run_font(r, bold=True)
    p_ttl = doc.add_paragraph()
    p_ttl.alignment = WD_ALIGN_PARAGRAPH.CENTER
    p_ttl.paragraph_format.first_line_indent = Cm(0)
    r_ttl = p_ttl.add_run(title)
    _set_run_font(r_ttl, bold=True)
    p_blank(doc)
    tbl = doc.add_table(rows=1, cols=len(headers))
    tbl.style = "Table Grid"
    for i, h in enumerate(headers):
        c = tbl.rows[0].cells[i]
        c.text = ""
        pp = c.paragraphs[0]
        pp.alignment = WD_ALIGN_PARAGRAPH.CENTER
        rr = pp.add_run(h)
        _set_run_font(rr, size=11, bold=True)
    for row in rows:
        cells = tbl.add_row().cells
        for i, t in enumerate(row):
            cells[i].text = ""
            pp = cells[i].paragraphs[0]
            rr = pp.add_run(t)
            _set_run_font(rr, size=11)
    p_blank(doc)
