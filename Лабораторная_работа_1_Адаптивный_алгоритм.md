#### ШАГ 2.1.2: Реализация алгоритма Simple Staircase (3 часа)
**СОЗДАЙТЕ НОВЫЙ ФАЙЛ** `eora/learning/adaptive_algorithms.py`:

```python
class SimpleStaircaseAdapter:
    def __init__(self, initial_difficulty, step_size, success_threshold, failure_threshold):
        self.initial_difficulty = initial_difficulty
        self.step_size = step_size
        self.success_threshold = success_threshold
        self.failure_threshold = failure_threshold
        
    def calculate_next_difficulty(self, current_difficulty, recent_attempts):
        """
        АЛГОРИТМ ПОШАГОВО:
        1. Проверить последовательность правильных ответов
        2. Проверить последовательность неправильных ответов
        3. Применить правила изменения сложности
        4. Ограничить результат диапазоном [0.0, 1.0]
        """
        
        if not recent_attempts:
            return self.initial_difficulty
            
        # ШАГ 1: Подсчет правильных ответов подряд с конца
        consecutive_correct = 0
        for attempt in reversed(recent_attempts):
            if attempt.is_correct:
                consecutive_correct += 1
            else:
                break
                
        # ШАГ 2: Подсчет неправильных ответов подряд с конца  
        consecutive_incorrect = 0
        for attempt in reversed(recent_attempts):
            if not attempt.is_correct:
                consecutive_incorrect += 1
            else:
                break
                
        # ШАГ 3: Применение правил
        new_difficulty = current_difficulty
        
        if consecutive_correct >= self.success_threshold:
            new_difficulty += self.step_size
            print(f"Повышение сложности: {consecutive_correct} правильных подряд")
            
        elif consecutive_incorrect >= self.failure_threshold:
            new_difficulty -= self.step_size
            print(f"Понижение сложности: {consecutive_incorrect} неправильных подряд")
            
        # ШАГ 4: Ограничение диапазона
        new_difficulty = max(0.0, min(1.0, new_difficulty))
        
        return new_difficulty
```

### Задание 2.2: Реализация Weighted Staircase Method

#### ШАГ 2.2.1: Реализация алгоритма (3 часа)
**ДОБАВЬТЕ В ФАЙЛ** `adaptive_algorithms.py`:

```python
class WeightedStaircaseAdapter:
    def __init__(self, initial_difficulty, step_size, target_accuracy=0.7, window_size=4):
        self.initial_difficulty = initial_difficulty
        self.step_size = step_size
        self.target_accuracy = target_accuracy
        self.window_size = window_size
        
    def calculate_next_difficulty(self, current_difficulty, recent_attempts, time_weights=None):
        """
        АЛГОРИТМ WEIGHTED STAIRCASE:
        1. Взять последние N попыток
        2. Присвоить им веса (новые попытки = больший вес)
        3. Посчитать взвешенную точность
        4. Скорректировать сложность к целевой точности
        """
        
        if not recent_attempts:
            return self.initial_difficulty
            
        # ШАГ 1: Берем последние попытки
        last_attempts = recent_attempts[-self.window_size:]
        
        # ШАГ 2: Веса по умолчанию (если не переданы)
        if time_weights is None:
            n = len(last_attempts)
            time_weights = [(i + 1) / n for i in range(n)]  # [0.25, 0.5, 0.75, 1.0] для 4 попыток
            
        # ШАГ 3: Расчет взвешенной точности
        weighted_sum = 0
        total_weight = 0
        
        for i, attempt in enumerate(last_attempts):
            weight = time_weights[i] if i < len(time_weights) else time_weights[-1]
            weighted_sum += weight * (1 if attempt.is_correct else 0)
            total_weight += weight
            
        weighted_accuracy = weighted_sum / total_weight if total_weight > 0 else 0.5
        
        # ШАГ 4: Коррекция сложности
        accuracy_diff = weighted_accuracy - self.target_accuracy
        adjustment = -accuracy_diff * self.step_size  # Если точность высокая -> увеличить сложность
        
        new_difficulty = current_difficulty + adjustment
        new_difficulty = max(0.0, min(1.0, new_difficulty))
        
        print(f"Взвешенная точность: {weighted_accuracy:.3f}, Целевая: {self.target_accuracy}, Коррекция: {adjustment:+.3f}")
        
        return new_difficulty
```

### Задание 2.3: Реализация упрощенного Bayesian Knowledge Tracing

#### ШАГ 2.3.1: Реализация BKT (3 часа)
**ДОБАВЬТЕ В ФАЙЛ** `adaptive_algorithms.py`:

```python
import math

class BayesianAdapter:
    def __init__(self, prior_knowledge=0.5, learning_rate=0.1, guess_probability=0.25, slip_probability=0.1):
        self.prior_knowledge = prior_knowledge      # P(L0) - начальная вероятность знания
        self.learning_rate = learning_rate          # P(T) - вероятность изучения при ошибке  
        self.guess_probability = guess_probability  # P(G) - вероятность угадывания
        self.slip_probability = slip_probability    # P(S) - вероятность ошибки при знании
        
    def update_knowledge_probability(self, current_prob, is_correct):
        """
        ФОРМУЛА БАЙЕСА ДЛЯ ОБНОВЛЕНИЯ ВЕРОЯТНОСТИ ЗНАНИЯ:
        
        P(L|evidence) = P(evidence|L) * P(L) / P(evidence)
        
        где:
        - P(L) - текущая вероятность знания
        - P(evidence|L) - вероятность наблюдаемого ответа при наличии знания
        - P(evidence) - общая вероятность наблюдаемого ответа
        """
        
        if is_correct:
            # Правильный ответ
            # P(correct|знает) = 1 - slip_probability
            # P(correct|не_знает) = guess_probability
            p_correct_if_knows = 1 - self.slip_probability
            p_correct_if_not_knows = self.guess_probability
            
            # P(correct) = P(correct|знает) * P(знает) + P(correct|не_знает) * P(не_знает)
            p_correct = p_correct_if_knows * current_prob + p_correct_if_not_knows * (1 - current_prob)
            
            # Обновляем вероятность знания
            new_prob = (p_correct_if_knows * current_prob) / p_correct
            
        else:
            # Неправильный ответ
            # P(incorrect|знает) = slip_probability  
            # P(incorrect|не_знает) = 1 - guess_probability
            p_incorrect_if_knows = self.slip_probability
            p_incorrect_if_not_knows = 1 - self.guess_probability
            
            # P(incorrect) = P(incorrect|знает) * P(знает) + P(incorrect|не_знает) * P(не_знает)
            p_incorrect = p_incorrect_if_knows * current_prob + p_incorrect_if_not_knows * (1 - current_prob)
            
            # Обновляем вероятность знания
            new_prob = (p_incorrect_if_knows * current_prob) / p_incorrect
            
            # После неправильного ответа есть шанс изучения
            new_prob = new_prob + (1 - new_prob) * self.learning_rate
            
        return min(0.99, max(0.01, new_prob))  # Ограничиваем диапазон
        
    def calculate_next_difficulty(self, current_difficulty, recent_attempts):
        """
        Преобразуем вероятность знания в уровень сложности:
        - Высокая вероятность знания -> высокая сложность
        - Низкая вероятность знания -> низкая сложность
        """
        knowledge_prob = self.prior_knowledge
        
        for attempt in recent_attempts:
            knowledge_prob = self.update_knowledge_probability(knowledge_prob, attempt.is_correct)
            
        # Простое преобразование: сложность = вероятность знания
        new_difficulty = knowledge_prob
        
        return new_difficulty
```

class VirtualStudent:
    def __init__(self, name, skill_level, consistency=0.8):
        """
        skill_level: множитель для сложности (0.3-1.1)
        consistency: насколько стабильно отвечает (0.5-1.0)
        """
        self.name = name
        self.skill_level = skill_level
        self.consistency = consistency
        
    def answer_task(self, task_difficulty):
        """
        Симуляция ответа на задачу заданной сложности
        """
        # Базовая вероятность правильного ответа
        base_probability = self.skill_level * (1 - task_difficulty) + 0.1
        base_probability = max(0.05, min(0.95, base_probability))
        
        # Добавляем случайность (непоследовательность)
        noise = random.uniform(-0.2, 0.2) * (1 - self.consistency)
        final_probability = max(0.05, min(0.95, base_probability + noise))
        
        # Генерируем случайный ответ
        return random.random() < final_probability
        
    def __str__(self):
        return f"{self.name} (навык: {self.skill_level}, стабильность: {self.consistency})"

import json

# Загружаем данные
with open('comparison_results.json', 'r', encoding='utf-8') as f:
    results = json.load(f)

def perform_t_tests():
    """
    Проводим парные t-тесты для сравнения алгоритмов
    """
    print("=== СТАТИСТИЧЕСКИЕ ТЕСТЫ (t-test) ===\n")
    
    algorithms = list(results.keys())
    
    # Собираем данные о времени достижения цели для каждого алгоритма
    time_data = {}
    accuracy_data = {}
    
    for algo in algorithms:
        times = []
        accuracies = []
        
        for student in results[algo].keys():
            times.append(results[algo][student]['avg_steps_to_target'])
            accuracies.append(results[algo][student]['avg_final_accuracy'])
        
        time_data[algo] = times
        accuracy_data[algo] = accuracies
    
    # ПАРНЫЕ СРАВНЕНИЯ ДЛЯ ВРЕМЕНИ ДОСТИЖЕНИЯ ЦЕЛИ
    print("1. СРАВНЕНИЕ ПО ВРЕМЕНИ ДОСТИЖЕНИЯ ЦЕЛИ")
    print("-" * 50)
    
    for i in range(len(algorithms)):
        for j in range(i+1, len(algorithms)):
            algo1, algo2 = algorithms[i], algorithms[j]
            
            # Проводим t-test
            t_stat, p_value = stats.ttest_rel(time_data[algo1], time_data[algo2])
            
            # Вычисляем средние значения
            mean1 = np.mean(time_data[algo1])
            mean2 = np.mean(time_data[algo2])
            
            print(f"{algo1} vs {algo2}:")
            print(f"  Средние значения: {mean1:.2f} vs {mean2:.2f}")
            print(f"  t-статистика: {t_stat:.3f}")
            print(f"  p-value: {p_value:.4f}")
            
            # Интерпретация результата
            alpha = 0.05
            if p_value < alpha:
                better = algo1 if mean1 < mean2 else algo2  # Меньше время = лучше
                print(f"  Результат: {better} ЗНАЧИМО ЛУЧШЕ (p < {alpha})")
            else:
                print(f"  Результат: НЕТ ЗНАЧИМОЙ РАЗНОСТИ (p >= {alpha})")
            print()
    
    # ПАРНЫЕ СРАВНЕНИЯ ДЛЯ ФИНАЛЬНОЙ ТОЧНОСТИ
    print("2. СРАВНЕНИЕ ПО ФИНАЛЬНОЙ ТОЧНОСТИ")  
    print("-" * 50)
    
    for i in range(len(algorithms)):
        for j in range(i+1, len(algorithms)):
            algo1, algo2 = algorithms[i], algorithms[j]
            
            t_stat, p_value = stats.ttest_rel(accuracy_data[algo1], accuracy_data[algo2])
            
            mean1 = np.mean(accuracy_data[algo1])
            mean2 = np.mean(accuracy_data[algo2])
            
            print(f"{algo1} vs {algo2}:")
            print(f"  Средние значения: {mean1:.3f} vs {mean2:.3f}")
            print(f"  t-статистика: {t_stat:.3f}")
            print(f"  p-value: {p_value:.4f}")
            
            if p_value < alpha:
                better = algo1 if mean1 > mean2 else algo2  # Больше точность = лучше
                print(f"  Результат: {better} ЗНАЧИМО ЛУЧШЕ (p < {alpha})")
            else:
                print(f"  Результат: НЕТ ЗНАЧИМОЙ РАЗНОСТИ (p >= {alpha})")
            print()

def calculate_correlations():
    """
    Рассчитываем корреляции между различными параметрами
    """
    print("=== КОРРЕЛЯЦИОННЫЙ АНАЛИЗ ===\n")
    
    # Собираем данные всех студентов и алгоритмов
    all_data = []

        'Слабый': 0.3,
        'Ниже среднего': 0.5,
        'Средний': 0.7,
        'Выше среднего': 0.9,
        'Сильный': 1.1
    }

    # Сохраняем корреляционную матрицу

# Пример расчета для Simple Staircase
def example_calculation():
    difficulty = 0.5
    step = 0.1
    threshold = 2
    
    answers = [True, True, False, True]  # T,T -> +step, F -> reset counter
    consecutive_correct = 0
    
    for answer in answers:
        if answer:
            consecutive_correct += 1
            if consecutive_correct >= threshold:
                difficulty = min(1.0, difficulty + step)
                consecutive_correct = 0
        else:
            difficulty = max(0.0, difficulty - step)
            consecutive_correct = 0
    
    return difficulty  # Ожидаемый результат: 0.4
