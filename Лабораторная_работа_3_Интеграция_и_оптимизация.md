```python
import requests
import json
from typing import Dict, List, Optional
from django.conf import settings
import logging

logger = logging.getLogger(__name__)

class [ВашПротокол]Adapter:
    """
    Адаптер для интеграции с [название системы из варианта]
    """
    
    def __init__(self):
        self.api_url = settings.[ПРОТОКОЛ]_API_URL
        self.api_key = settings.[ПРОТОКОЛ]_API_KEY
        self.timeout = 30
        self.retry_count = 3
    
    def authenticate(self) -> str:
        """
        Получение токена аутентификации
        """
        auth_url = f"{self.api_url}/auth"
        
        try:
            response = requests.post(
                auth_url,
                json={
                    'api_key': self.api_key,
                    'grant_type': 'client_credentials'
                },
                timeout=self.timeout
            )
            response.raise_for_status()
            
            return response.json()['access_token']
            
        except requests.RequestException as e:
            logger.error(f"Ошибка аутентификации: {e}")
            raise IntegrationError(f"Не удалось аутентифицироваться: {e}")
    
    def export_course(self, course_data: Dict) -> str:
        """
        Экспорт курса в формате внешней системы
        РЕАЛИЗУЙТЕ ДЛЯ ВАШЕГО ПРОТОКОЛА
        """
        token = self.authenticate()
        
        # Преобразуем данные EORA в формат внешней системы
        formatted_data = self._format_course_data(course_data)
        
        headers = {
            'Authorization': f'Bearer {token}',
            'Content-Type': 'application/json'
        }
        
        try:
            response = requests.post(
                f"{self.api_url}/courses",
                json=formatted_data,
                headers=headers,
                timeout=self.timeout
            )
            response.raise_for_status()
            
            return response.json()['course_id']
            
        except requests.RequestException as e:
            logger.error(f"Ошибка экспорта курса: {e}")
            raise IntegrationError(f"Не удалось экспортировать курс: {e}")
    
    def import_results(self, course_id: str) -> List[Dict]:
        """
        Импорт результатов обучения из внешней системы
        """
        token = self.authenticate()
        
        headers = {'Authorization': f'Bearer {token}'}
        
        try:
            response = requests.get(
                f"{self.api_url}/courses/{course_id}/results",
                headers=headers,
                timeout=self.timeout
            )
            response.raise_for_status()
            
            return self._parse_results(response.json())
            
        except requests.RequestException as e:
            logger.error(f"Ошибка импорта результатов: {e}")
            raise IntegrationError(f"Не удалось импортировать результаты: {e}")
    
    def _format_course_data(self, course_data: Dict) -> Dict:
        """
        Преобразование данных курса EORA в формат внешней системы
        АДАПТИРУЙТЕ ПОД ВАШ ПРОТОКОЛ
        """
        formatted = {
            'title': course_data['title'],
            'description': course_data['description'],
            'activities': []
        }
        
        # Преобразуем задачи EORA в активности внешней системы
        for task in course_data.get('tasks', []):
            activity = {
                'id': str(task['id']),
                'type': 'assessment',
                'title': task['question'],
                'max_score': 100,
                'metadata': {
                    'difficulty': task['difficulty'],
                    'category': task['category']
                }
            }
            formatted['activities'].append(activity)
        
        return formatted
    
    def _parse_results(self, raw_results: Dict) -> List[Dict]:
        """
        Парсинг результатов из внешней системы в формат EORA
        """
        parsed_results = []
        
        for result in raw_results.get('results', []):
            parsed = {
                'user_id': result['learner_id'],
                'task_id': int(result['activity_id']),
                'score': result['score'],
                'completion_time': result['duration'],
                'timestamp': result['completed_at']
            }
            parsed_results.append(parsed)
        
        return parsed_results
```

#### ШАГ 1.2.2: Создание API для управления интеграциями (2 часа)

```python
from rest_framework import status
from rest_framework.decorators import api_view, permission_classes
from rest_framework.permissions import IsAuthenticated
from rest_framework.response import Response
from .adapters import IntegrationManager, IntegrationError

@api_view(['POST'])
@permission_classes([IsAuthenticated])
def sync_course(request, course_id):
    """
    Синхронизация курса с внешней системой
    """
    protocol = request.data.get('protocol')
    if not protocol:
        return Response({
            'error': 'Не указан протокол интеграции'
        }, status=status.HTTP_400_BAD_REQUEST)
    
    try:
        manager = IntegrationManager()
        result = manager.sync_course_data(course_id, protocol)
        
        return Response(result)
        
    except IntegrationError as e:
        return Response({
            'error': str(e)
        }, status=status.HTTP_502_BAD_GATEWAY)
        
    except Exception as e:
        return Response({
            'error': 'Внутренняя ошибка сервера'
        }, status=status.HTTP_500_INTERNAL_SERVER_ERROR)

@api_view(['GET'])
@permission_classes([IsAuthenticated])
def integration_status(request, course_id):
    """
    Статус интеграции курса
    """
    # Реализуйте проверку статуса

                'last_sync': '2024-01-15T10:30:00Z',
```

```python
class CacheManager:
    """
    Менеджер кэширования для системы EORA
    Реализация согласно варианту задания
    """
    
    def __init__(self):
        self.default_cache = caches['default']
        self.analytics_cache = caches['analytics']
        self.session_cache = caches['sessions']

        self.analytics_ttl = settings.CACHE_SETTINGS['ANALYTICS_CACHE_TTL']
        self.user_ttl = settings.CACHE_SETTINGS['USER_CACHE_TTL']
        self.task_ttl = settings.CACHE_SETTINGS['TASK_CACHE_TTL']
    
    def get_cache_key(self, prefix: str, *args, **kwargs) -> str:
        """
        Генерация уникального ключа кэша
        """
        key_parts = [prefix] + list(map(str, args))
        
        if kwargs:
            sorted_kwargs = sorted(kwargs.items())
            key_parts.extend(f"{k}:{v}" for k, v in sorted_kwargs)
        
        key_string = ":".join(key_parts)
        
        # Хэшируем длинные ключи
        if len(key_string) > 200:
            key_hash = hashlib.md5(key_string.encode()).hexdigest()
            return f"{prefix}:hash:{key_hash}"
        
        return key_string

        """
        Кэширование аналитических данных согласно варианту
        """

        # Добавляем метаданные
        cached_data = {
            'data': data,
            'timestamp': timezone.now().isoformat(),
            'user_id': user_id,
            'period': period,
            'version': settings.CACHE_SETTINGS['CACHE_VERSION']
        }
        
        try:
            self.analytics_cache.set(
                cache_key, 
                cached_data, 
                timeout=self.analytics_ttl
            )
            logger.info(f"Кэшированы аналитические данные: {cache_key}")
            
        except Exception as e:

def cache_result(cache_key_prefix: str, timeout: int = 3600):
    """
    Декоратор для автоматического кэширования результатов функций
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_manager = CacheManager()
            
            # Генерируем ключ кэша на основе аргументов функции
            cache_key = cache_manager.get_cache_key(
                cache_key_prefix, 
                func.__name__, 
                *args, 
                **kwargs
            )
            
            # Пытаемся получить из кэша
            result = cache_manager.default_cache.get(cache_key)
            
            if result is None:
                # Вычисляем результат и кэшируем
                result = func(*args, **kwargs)
                cache_manager.default_cache.set(cache_key, result, timeout)
                logger.info(f"Результат функции {func.__name__} кэширован")
            else:
                logger.info(f"Результат функции {func.__name__} получен из кэша")
            
            return result
        
        return wrapper
    return decorator
```

```python
import time
from django.core.cache import caches
from django.conf import settings

logger = logging.getLogger(__name__)

class CacheMonitor:
    """
    Мониторинг производительности и состояния кэша
    """
    
    def __init__(self):
        self.redis_client = redis.Redis(
            host='127.0.0.1', 
            port=6379, 
            db=1,
            decode_responses=True
        )
    
    def get_cache_stats(self) -> dict:
        """
        Получение статистики кэша Redis
        """
        try:
            info = self.redis_client.info()
            
            stats = {
                'connected_clients': info.get('connected_clients', 0),
                'used_memory': info.get('used_memory_human', '0B'),
                'used_memory_peak': info.get('used_memory_peak_human', '0B'),
                'keyspace_hits': info.get('keyspace_hits', 0),
                'keyspace_misses': info.get('keyspace_misses', 0),
                'expired_keys': info.get('expired_keys', 0),
                'evicted_keys': info.get('evicted_keys', 0)
            }
            
            # Рассчитываем hit ratio
            hits = stats['keyspace_hits']
            misses = stats['keyspace_misses']
            total = hits + misses
            
            if total > 0:
                stats['hit_ratio'] = round((hits / total) * 100, 2)
            else:
                stats['hit_ratio'] = 0
            
            return stats
            
        except Exception as e:
            logger.error(f"Ошибка получения статистики кэша: {e}")
            return {}
    
    def measure_cache_performance(self, operations: int = 1000) -> dict:
        """
        Измерение производительности кэша
        """
        cache = caches['default']
        
        # Тест записи
        write_times = []
        for i in range(operations):
            start_time = time.time()
            cache.set(f'perf_test:{i}', f'test_data_{i}', timeout=60)
            write_times.append(time.time() - start_time)
        
        # Тест чтения
        read_times = []
        for i in range(operations):
            start_time = time.time()
            cache.get(f'perf_test:{i}')
            read_times.append(time.time() - start_time)
        
        # Очистка тестовых данных
        for i in range(operations):
            cache.delete(f'perf_test:{i}')
        
        return {
            'operations_count': operations,
            'avg_write_time': sum(write_times) / len(write_times),
            'avg_read_time': sum(read_times) / len(read_times),
            'max_write_time': max(write_times),
            'max_read_time': max(read_times),
            'total_time': sum(write_times) + sum(read_times)
        }
    
    def check_cache_health(self) -> dict:
        """
        Проверка здоровья системы кэширования
        """
        health_status = {
            'status': 'healthy',
            'issues': []
        }
        
        try:
            # Проверяем подключение
            self.redis_client.ping()
            
            # Проверяем статистику
            stats = self.get_cache_stats()
            
            # Проверяем hit ratio (должен быть > 70% для варианта)
            if stats.get('hit_ratio', 0) < 70:
                health_status['issues'].append(
                    f"Низкий hit ratio: {stats.get('hit_ratio', 0)}%"
                )
                health_status['status'] = 'warning'
            
            # Проверяем использование памяти
            memory_info = self.redis_client.info('memory')
            used_memory_pct = memory_info.get('used_memory_rss', 0) / (1024**3)  # GB
            
            if used_memory_pct > 0.8:  # 80% памяти
                health_status['issues'].append(
                    f"Высокое использование памяти: {used_memory_pct:.1f}GB"
                )
                health_status['status'] = 'critical'
            
            # Проверяем количество вытесненных ключей
            if stats.get('evicted_keys', 0) > 1000:
                health_status['issues'].append(
                    f"Много вытесненных ключей: {stats.get('evicted_keys')}"
                )
                health_status['status'] = 'warning'
            
        except Exception as e:
            health_status['status'] = 'critical'
            health_status['issues'].append(f"Ошибка подключения к Redis: {e}")
        
        return health_status
```

```python
class DatabaseProfiler:
    """
    Профилировщик производительности базы данных
    """
    
    def __init__(self):
        self.queries_log = []
    
    def profile_queries(self):
        """
        Контекстный менеджер для профилирования запросов
        """
        return QueryProfilerContext(self)
    
    def analyze_slow_queries(self, threshold_ms: float = 100) -> list:
        """
        Анализ медленных запросов
        """
        slow_queries = []
        
        for query_info in self.queries_log:
            if query_info['time'] > threshold_ms / 1000:  # конвертируем в секунды
                slow_queries.append(query_info)
        
        # Сортируем по времени выполнения
        slow_queries.sort(key=lambda x: x['time'], reverse=True)
        
        return slow_queries
    
    def get_optimization_suggestions(self) -> dict:
        """
        Предложения по оптимизации на основе анализа запросов
        """
        suggestions = {
            'missing_indexes': [],
            'n_plus_one_queries': [],
            'complex_queries': []
        }
        
        # Анализируем паттерны запросов
        query_patterns = {}
        for query_info in self.queries_log:
            sql = query_info['sql']
            
            # Упрощаем запрос для группировки
            simplified = self._simplify_query(sql)
            
            if simplified in query_patterns:
                query_patterns[simplified]['count'] += 1
            else:
                query_patterns[simplified] = {
                    'count': 1,
                    'example': sql,
                    'avg_time': query_info['time']
                }
        
        # Находим потенциальные N+1 проблемы
        for pattern, info in query_patterns.items():
            if info['count'] > 10 and 'SELECT' in pattern and 'WHERE' in pattern:
                suggestions['n_plus_one_queries'].append({
                    'pattern': pattern,
                    'count': info['count'],
                    'suggestion': 'Рассмотрите использование select_related() или prefetch_related()'
                })
        
        return suggestions

class QueryProfilerContext:
    def __init__(self, profiler):
        self.profiler = profiler
        self.initial_queries_count = 0

        self.initial_queries_count = len(connection.queries)
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):

        new_queries = connection.queries[self.initial_queries_count:]
        
        for query in new_queries:
            self.profiler.queries_log.append({
                'sql': query['sql'],
                'time': float(query['time']),
                'timestamp': time.time()
            })
```

```python
                        default=Value(0),
                        output_field=FloatField()
                    )

            )
```

```python
# monitoring/performance_monitor.py
import psutil
import time
import threading
from django.core.cache import cache
from django.db import connection
import logging

logger = logging.getLogger(__name__)

class PerformanceMonitor:
    """
    Мониторинг производительности системы во время нагрузочных тестов
    """
    
    def __init__(self):
        self.monitoring = False
        self.metrics_history = []
        self.alert_thresholds = {
            'cpu_percent': 80,      # % CPU
            'memory_percent': 85,   # % RAM  
            'response_time': 500,   # ms
            'error_rate': 5,        # %
            'db_connections': 90    # максимальное количество соединений
        }
    
    def start_monitoring(self, interval: int = 5):
        """
        Запуск мониторинга с заданным интервалом
        """
        self.monitoring = True
        monitor_thread = threading.Thread(
            target=self._monitoring_loop, 
            args=(interval,)
        )
        monitor_thread.daemon = True
        monitor_thread.start()
        
        logger.info(f"Мониторинг запущен с интервалом {interval} секунд")
    
    def stop_monitoring(self):
        """
        Остановка мониторинга
        """
        self.monitoring = False
        logger.info("Мониторинг остановлен")
    
    def _monitoring_loop(self, interval: int):
        """
        Основной цикл мониторинга
        """
        while self.monitoring:
            try:
                metrics = self._collect_metrics()
                self.metrics_history.append(metrics)
                
                # Проверяем пороговые значения
                self._check_alerts(metrics)
                
                # Ограничиваем размер истории
                if len(self.metrics_history) > 1000:
                    self.metrics_history = self.metrics_history[-500:]
                
                time.sleep(interval)
                
            except Exception as e:
                logger.error(f"Ошибка мониторинга: {e}")
    
    def _collect_metrics(self) -> dict:
        """
        Сбор метрик производительности
        """
        # Системные метрики
        cpu_percent = psutil.cpu_percent(interval=1)
        memory = psutil.virtual_memory()
        disk = psutil.disk_usage('/')
        
        # Метрики базы данных
        db_metrics = self._get_database_metrics()
        
        # Метрики кэша
        cache_metrics = self._get_cache_metrics()
        
        metrics = {
            'timestamp': time.time(),
            'system': {
                'cpu_percent': cpu_percent,
                'memory_percent': memory.percent,
                'memory_used_gb': memory.used / (1024**3),
                'disk_percent': disk.percent,
                'load_average': psutil.getloadavg()[0] if hasattr(psutil, 'getloadavg') else 0
            },
            'database': db_metrics,
            'cache': cache_metrics,
            'application': self._get_application_metrics()
        }
        
        return metrics
    
    def _get_database_metrics(self) -> dict:
        """
        Метрики базы данных PostgreSQL/SQLite
        """
        try:
            with connection.cursor() as cursor:
                # Количество активных соединений
                cursor.execute("""
                    SELECT count(*) FROM pg_stat_activity 
                    WHERE state = 'active'
                """)
                active_connections = cursor.fetchone()[0]
                
                # Медленные запросы (> 1 секунды)
                cursor.execute("""
                    SELECT count(*) FROM pg_stat_activity 
                    WHERE state = 'active' 
                    AND now() - query_start > interval '1 second'
                """)
                slow_queries = cursor.fetchone()[0]
                
            return {
                'active_connections': active_connections,
                'slow_queries': slow_queries,
                'total_queries': len(connection.queries)
            }
        except Exception as e:
            logger.error(f"Ошибка сбора метрик БД: {e}")
            return {
                'active_connections': 0,
                'slow_queries': 0,
                'total_queries': len(connection.queries)
            }
    
    def _get_cache_metrics(self) -> dict:
        """
        Метрики кэша Redis
        """
        try:
            from caching.monitoring import CacheMonitor
            monitor = CacheMonitor()
            return monitor.get_cache_stats()
        except Exception as e:
            logger.error(f"Ошибка сбора метрик кэша: {e}")
            return {}
    
    def _get_application_metrics(self) -> dict:
        """
        Метрики Django приложения
        """
        return {
            'active_sessions': cache.get('active_sessions_count', 0),
            'requests_per_minute': cache.get('requests_per_minute', 0),
            'error_rate': cache.get('error_rate', 0)
        }
    
    def _check_alerts(self, metrics: dict):
        """
        Проверка пороговых значений и отправка алертов
        """
        alerts = []
        
        # Проверка CPU
        if metrics['system']['cpu_percent'] > self.alert_thresholds['cpu_percent']:
            alerts.append(f"Высокая загрузка CPU: {metrics['system']['cpu_percent']:.1f}%")
        
        # Проверка памяти
        if metrics['system']['memory_percent'] > self.alert_thresholds['memory_percent']:
            alerts.append(f"Высокое использование памяти: {metrics['system']['memory_percent']:.1f}%")
        
        # Проверка соединений с БД
        db_conn = metrics['database'].get('active_connections', 0)
        if db_conn > self.alert_thresholds['db_connections']:
            alerts.append(f"Много активных соединений с БД: {db_conn}")
        
        # Отправляем алерты
        if alerts:
            for alert in alerts:
                logger.warning(f"ALERT: {alert}")
                # Здесь можно добавить отправку email/Slack уведомлений
    
    def generate_report(self) -> dict:
        """
        Генерация отчета по результатам мониторинга
        """
        if not self.metrics_history:
            return {'error': 'Нет данных для отчета'}
        
        # Агрегируем данные
        cpu_values = [m['system']['cpu_percent'] for m in self.metrics_history]
        memory_values = [m['system']['memory_percent'] for m in self.metrics_history]
        
        report = {
            'test_duration': len(self.metrics_history),
            'system_performance': {
                'avg_cpu': sum(cpu_values) / len(cpu_values),
                'max_cpu': max(cpu_values),
                'avg_memory': sum(memory_values) / len(memory_values),
                'max_memory': max(memory_values)
            },
            'alerts_count': sum(1 for m in self.metrics_history 
                              if m['system']['cpu_percent'] > self.alert_thresholds['cpu_percent']),
            'performance_grade': self._calculate_performance_grade()
        }
        
        return report
    
    def _calculate_performance_grade(self) -> str:
        """
        Оценка производительности системы
        """
        if not self.metrics_history:
            return 'N/A'
        
        # Простая система оценки на основе превышения порогов
        violations = 0
        total_checks = len(self.metrics_history)
        
        for metrics in self.metrics_history:
            if metrics['system']['cpu_percent'] > self.alert_thresholds['cpu_percent']:
                violations += 1
            if metrics['system']['memory_percent'] > self.alert_thresholds['memory_percent']:
                violations += 1
        
        violation_rate = violations / (total_checks * 2)  # 2 проверки на метрику
        
        if violation_rate < 0.05:  # < 5% нарушений
            return 'Отлично'
        elif violation_rate < 0.15:  # < 15% нарушений
            return 'Хорошо'
        elif violation_rate < 0.3:  # < 30% нарушений
            return 'Удовлетворительно'
        else:
            return 'Требует оптимизации'

        report = monitor.generate_report()
```
