```python

from django.test import TestCase
from django.contrib.auth.models import User
from .models import Course, Task, LearningSession

class CourseModelTest(TestCase):
    """Тесты для модели Course"""
    
    def setUp(self):
        """Настройка перед каждым тестом"""
        self.course = Course.objects.create(
            title='Тестовый курс',
            description='Описание курса'
        )
    
    def test_course_creation(self):
        """Тест создания курса"""
        self.assertEqual(self.course.title, 'Тестовый курс')
        self.assertIsNotNone(self.course.created_at)
    
    def test_course_str(self):
        """Тест метода __str__"""
        self.assertEqual(str(self.course), 'Тестовый курс')

class TaskModelTest(TestCase):
    """Тесты для модели Task"""
    
    def setUp(self):
        self.course = Course.objects.create(title='Курс')
        self.task = Task.objects.create(
            course=self.course,
            title='Задача',
            text='Условие',
            correct_answer=5.0,
            difficulty=3
        )
    
    def test_task_creation(self):
        """Тест создания задачи"""
        self.assertEqual(self.task.title, 'Задача')
        self.assertEqual(self.task.difficulty, 3)
    
    def test_check_answer_correct(self):
        """Тест проверки правильного ответа"""
        self.assertTrue(self.task.check_answer(5.0))
    
    def test_check_answer_wrong(self):
        """Тест проверки неправильного ответа"""
        self.assertFalse(self.task.check_answer(4.0))
    
    def test_task_course_relation(self):
        """Тест связи задача → курс"""
        self.assertEqual(self.task.course, self.course)
        self.assertIn(self.task, self.course.tasks.all())
```

**Требования:**
- ✅ Минимум 5 тестов для разных моделей
- ✅ Использование `setUp()` для подготовки данных
- ✅ Понятные названия тестов с docstring
- ✅ Тесты проверяют реальную функциональность

### 2. Тестирование API

#### 2.1. Тесты для API эндпоинтов

Создайте минимум **5 тестов** для ваших API эндпоинтов:

**Требования:**
- Тест получения списка (GET)
- Тест создания объекта (POST)
- Тест получения деталей (GET /id/)
- Тест обновления (PATCH или PUT)
- Тест удаления (DELETE) или кастомного действия

**Пример:**
```python
from rest_framework.test import APIClient
from rest_framework import status

class CourseAPITest(TestCase):
    def setUp(self):
        self.client = APIClient()
        self.user = User.objects.create_user(
            username='testuser',
            password='testpass123'
        )
        self.client.force_authenticate(user=self.user)
    
    def test_list_courses(self):
        """Тест получения списка курсов"""
        Course.objects.create(title='Курс 1')
        Course.objects.create(title='Курс 2')
        
        response = self.client.get('/api/courses/')
        
        self.assertEqual(response.status_code, status.HTTP_200_OK)
        self.assertEqual(len(response.data), 2)
    
    def test_create_course(self):
        """Тест создания курса"""
        data = {
            'title': 'Новый курс',
            'description': 'Описание'
        }
        
        response = self.client.post('/api/courses/', data, format='json')
        
        self.assertEqual(response.status_code, status.HTTP_201_CREATED)
        self.assertEqual(Course.objects.count(), 1)
        self.assertEqual(Course.objects.get().title, 'Новый курс')
    
    def test_create_course_validation(self):
        """Тест валидации при создании"""
        data = {
            'title': '',  # Пустое название
        }
        
        response = self.client.post('/api/courses/', data, format='json')
        
        self.assertEqual(response.status_code, status.HTTP_400_BAD_REQUEST)
    
    def test_get_course_detail(self):
        """Тест получения деталей курса"""
        course = Course.objects.create(title='Курс')
        
        response = self.client.get(f'/api/courses/{course.id}/')
        
        self.assertEqual(response.status_code, status.HTTP_200_OK)
        self.assertEqual(response.data['title'], 'Курс')
    
    def test_update_course(self):
        """Тест обновления курса"""
        course = Course.objects.create(title='Старое название')
        
        data = {'title': 'Новое название'}
        response = self.client.patch(
            f'/api/courses/{course.id}/',
            data,
            format='json'
        )
        
        self.assertEqual(response.status_code, status.HTTP_200_OK)
        course.refresh_from_db()
        self.assertEqual(course.title, 'Новое название')
```

**Требования:**
- ✅ Минимум 5 тестов для разных эндпоинтов
- ✅ Тесты проверяют успешные сценарии
- ✅ Тесты проверяют ошибки (валидация, 404, 403)
- ✅ Использование `APIClient` и `force_authenticate`

```python
        self.user = User.objects.create_user(
            username='user',
            password='pass123'
        )
    
    def test_unauthorized_access(self):
        """Тест доступа без авторизации"""

        self.client.force_authenticate(user=self.user)
        response = self.client.get('/api/courses/')
```

```python
for course in courses:
    print(course.author.username)  # Данные уже загружены
```

```python
from django.test import TestCase
from django.contrib.auth.models import User
from rest_framework.test import APIClient
from rest_framework import status
from .models import Course, Task, LearningSession

class CourseModelTest(TestCase):
    def setUp(self):
        self.course = Course.objects.create(
            title='Тестовый курс',
            description='Описание'
        )
    
    def test_course_creation(self):
        self.assertEqual(self.course.title, 'Тестовый курс')
        self.assertIsNotNone(self.course.created_at)
    
    def test_course_str(self):
        self.assertEqual(str(self.course), 'Тестовый курс')

class TaskModelTest(TestCase):
    def setUp(self):
        self.course = Course.objects.create(title='Курс')
        self.task = Task.objects.create(
            course=self.course,
            title='Задача',
            text='Условие',
            correct_answer=5.0
        )
    
    def test_check_answer_correct(self):
        self.assertTrue(self.task.check_answer(5.0))
    
    def test_check_answer_wrong(self):
        self.assertFalse(self.task.check_answer(4.0))

class CourseAPITest(TestCase):
    def setUp(self):
        self.client = APIClient()
        self.user = User.objects.create_user(
            username='user',
            password='pass123'
        )
        self.client.force_authenticate(user=self.user)
    
    def test_list_courses(self):
        Course.objects.create(title='Курс 1')
        Course.objects.create(title='Курс 2')
        
        response = self.client.get('/api/courses/')
        
        self.assertEqual(response.status_code, status.HTTP_200_OK)
        self.assertEqual(len(response.data), 2)
    
    def test_create_course(self):
        data = {'title': 'Новый курс'}
        response = self.client.post('/api/courses/', data, format='json')
        
        self.assertEqual(response.status_code, status.HTTP_201_CREATED)
        self.assertEqual(Course.objects.count(), 1)

class AuthenticationTest(TestCase):
    def setUp(self):
        self.client = APIClient()
        self.user = User.objects.create_user(
            username='user',
            password='pass123'
        )
    
    def test_unauthorized_access(self):
        response = self.client.get('/api/courses/')
        self.assertEqual(response.status_code, status.HTTP_401_UNAUTHORIZED)
    
    def test_authorized_access(self):
        self.client.force_authenticate(user=self.user)
        response = self.client.get('/api/courses/')
        self.assertEqual(response.status_code, status.HTTP_200_OK)
```

## Рекомендации

1. **Пишите тесты параллельно с кодом** — не откладывайте на потом
2. **Начните с простых тестов** — сначала тесты моделей, потом API
3. **Используйте setUp()** — для подготовки данных
4. **Тестируйте граничные случаи** — 0, отрицательные значения, пустые строки
5. **Оптимизируйте постепенно** — найдите узкие места, потом оптимизируйте

## Часто задаваемые вопросы

**Q: Сколько тестов нужно написать?**  
A: Минимум 7 тестов (5 для моделей + минимум 2 для API, или наоборот). Чем больше, тем лучше.

**Q: Как проверить, что тесты работают?**  
A: Запустите `python manage.py test` — все тесты должны пройти без ошибок.

**Q: Нужно ли тестировать все методы?**  
A: Нет, достаточно протестировать критичные части: модели, API, бизнес-логику.

**Q: Как найти места для оптимизации?**  
A: Используйте Django Debug Toolbar или логируйте SQL-запросы. Ищите места с множественными запросами к БД.

**Q: Обязательно ли деплоить на сервер?**  
A: Нет, достаточно подготовить проект к production (настройки, документация). Деплой опционален.
